1. PowerDNS has been deployed!

2. Get the DNS service IP:
{{- if contains "NodePort" .Values.service.dns.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "powerdns.fullname" . }}-dns)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.dns.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "powerdns.fullname" . }}-dns'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "powerdns.fullname" . }}-dns --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "DNS Server IP: $SERVICE_IP:53"
{{- else if contains "ClusterIP" .Values.service.dns.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "powerdns.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 5353:$CONTAINER_PORT
  echo "DNS Server available at: 127.0.0.1:5353"
{{- end }}

{{- if .Values.powerdns.api.enabled }}

3. Access the PowerDNS API:
{{- if .Values.powerdns.webserver.enabled }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "powerdns.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8081:{{ .Values.powerdns.webserver.port }}
  echo "PowerDNS API available at: http://127.0.0.1:8081/"

  # Get the API key:
  kubectl get secret --namespace {{ .Release.Namespace }} {{ .Values.powerdns.api.secretName }} -o jsonpath="{.data.{{ .Values.powerdns.api.secretKey }}}" | base64 --decode; echo
{{- end }}
{{- end }}

4. Test DNS resolution:
   dig @$SERVICE_IP test-cluster.agentydragon.com

{{- if .Values.externalSecret.enabled }}

5. External Secret status:
   kubectl get externalsecret --namespace {{ .Release.Namespace }} {{ include "powerdns.fullname" . }}-api-key
{{- else }}

5. Note: External Secret integration is disabled.
   To enable automatic API key management from Vault:
   --set externalSecret.enabled=true
{{- end }}
